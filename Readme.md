CREATE TABLE users (
id BIGINT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(255) NOT NULL,
external_id VARCHAR(255),
date_created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
activated BOOLEAN NOT NULL DEFAULT FALSE
);
